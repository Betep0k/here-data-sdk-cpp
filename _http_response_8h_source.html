128 HttpResponse(const HttpResponse& other)
129 : status(other.status), headers(other.headers) {
130 response << other.response.rdbuf();
131 if (!response.good()) {
132 // Depending on the users handling of the stringstream it might be that
133 // the read position is already at the end and thus operator<< cannot
134 // read anything, so lets try with the safer but more memory intensive
135 // solution as a second step.
136 response.str(other.response.str());
137 }
138 }

149 if (this != &other) {
