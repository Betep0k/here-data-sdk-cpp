415 template <typename Key, typename Value, typename CacheCostFunc,
416  typename Compare, template <typename> class Alloc>
417 template <typename _Key, typename _Value>
418 auto LruCache<Key, Value, CacheCostFunc, Compare, Alloc>::Insert(_Key&& key,
419  _Value&& value)
420  -> std::pair<const_iterator, bool> {
421  Bucket bucket{map_.end(), map_.end(), std::forward<_Value>(value)};
422 
423  // item too large, do not insert
424  std::size_t valueCost = cache_cost_func_(bucket.value_);
425  if (valueCost > max_size_)
426  return std::make_pair(const_iterator{end()}, false);
427 
428  auto it =
429  porting::try_emplace(map_, std::forward<_Key>(key), std::move(bucket));
430 
431  if (it.second) {
432  AddInternal(it.first, valueCost);
433  return std::make_pair(const_iterator{it.first}, true);
434  }
435  Promote(it.first);
436  return std::make_pair(const_iterator{it.first}, false);
437 }

439 template <typename Key, typename Value, typename CacheCostFunc,
440  typename Compare, template <typename> class Alloc>
441 template <typename _Value>
442 auto LruCache<Key, Value, CacheCostFunc, Compare, Alloc>::InsertOrAssign(
443  Key key, _Value&& value) -> std::pair<const_iterator, bool> {
444  // as we need "key" with the right type a couple of times, create a temp
445  // copy to prevent implicit conversions below
446  auto it = map_.lower_bound(key);
447  if (it != map_.end() && IsEqual(key, it->first)) {
448  // element already exists, update it
449  std::size_t oldCost = cache_cost_func_(it->second.value_);
450  it->second.value_ = std::forward<_Value>(value);
451  std::size_t newCost = cache_cost_func_(it->second.value_);
452  AddInternal(it, newCost, &oldCost);
453  return std::make_pair(const_iterator{it}, false);
454  } else {
455  // element doesn't exist, insert it
456  Bucket bucket{map_.end(), map_.end(), std::forward<_Value>(value)};
457  std::size_t newCost = cache_cost_func_(bucket.value_);
458  if (newCost > max_size_)
459  return std::make_pair(const_iterator{end()}, false);
460 
461  auto new_it =
462  map_.insert(it, std::make_pair(std::move(key), std::move(bucket)));
463  AddInternal(new_it, newCost);
464  return std::make_pair(const_iterator{new_it}, true);
465  }
466 }

468 template <typename Key, typename Value, typename CacheCostFunc,
469  typename Compare, template <typename> class Alloc>
470 void LruCache<Key, Value, CacheCostFunc, Compare, Alloc>::Promote(
471  const typename MapType::iterator& it) {
472  if (it == first_)
473  return; // nothing to do
474 
475  // re-link previous and next buckets together
476  it->second.previous_->second.setNext(it->second.next_);
477  if (it->second.next_ != map_.end())
478  it->second.next_->second.setPrevious(it->second.previous_);
479  else
480  last_ = it->second.previous_;
481 
482  // re-link our bucket
483  it->second.setPrevious(map_.end());
484  it->second.setNext(first_);
485 
486  // update current head to ourselves
487  first_->second.setPrevious(it);
488  first_ = it;
489 }

491 template <typename Key, typename Value, typename CacheCostFunc,
492  typename Compare, template <typename> class Alloc>
493 void LruCache<Key, Value, CacheCostFunc, Compare, Alloc>::AddInternal(
494  const typename MapType::iterator& it, std::size_t cost,
495  std::size_t* oldCost) {
496  if (!oldCost) {
497  // new bucket added
498  if (map_.size() == 1) {
499  // we're the first and only one.
500  assert(last_ == map_.end() && first_ == map_.end());
501  last_ = first_ = it;
502  } else {
503  // relink first bucket
504  it->second.setNext(first_);
505  first_->second.setPrevious(it);
506  first_ = it;
507  }
508  size_ += cost;
509  } else {
510  // key already in map - only promote
511  size_ += cost - *oldCost;
512  Promote(it);
513  }
514 
515  evict();
516 }

518 template <typename Key, typename Value, typename CacheCostFunc,
519  typename Compare, template <typename> class Alloc>
520 void LruCache<Key, Value, CacheCostFunc, Compare, Alloc>::Erase(
521  typename MapType::iterator it, bool doEvictionCallback) {
522  std::size_t cost = cache_cost_func_(it->second.value_);
523 
524  if (it->second.next_ == map_.end())
525  last_ = it->second.previous_;
526  else
527  it->second.next_->second.setPrevious(it->second.previous_);
528 
529  if (it->second.previous_ == map_.end())
530  first_ = it->second.next_;
531  else
532  it->second.previous_->second.setNext(it->second.next_);
533 
534  if (doEvictionCallback && eviction_callback_)
535  eviction_callback_(it->first, std::move(it->second.value_));
536 
537  map_.erase(it);
538  size_ -= cost;
539 }
