333 typename AlignedBox<T, N>::VectorType AlignedBox<T, N>::Center() const {
334 #ifndef NDEBUG
335  // Check for possible 64-bit integer overflow/underflow
336  if (std::numeric_limits<OverflowType>::is_integer &&
337  sizeof(OverflowType) * 8 == 64) {
338  for (unsigned dim = 0; dim < dimensions; ++dim) {

426 template <typename T, unsigned int N>
427 auto AlignedBox<T, N>::Distance2(const VectorType& point) const -> ValueType {
428  T distance2 = 0;
429  for (unsigned dim = 0; dim < dimensions; dim++) {
430  if (point[dim] < minimum_[dim]) {
431  T dimDistance = minimum_[dim] - point[dim];
432  distance2 += dimDistance * dimDistance;
433  } else if (point[dim] > maximum_[dim]) {
434  T dimDistance = point[dim] - maximum_[dim];
435  distance2 += dimDistance * dimDistance;
436  }
437  }
438 
439  return distance2;
440 }
