51template <uint64_t MinimumValidity>
52class TokenProvider {
53 public:
60 explicit TokenProvider(Settings settings)

108 class TokenProviderImpl {
109 public:
110 static constexpr auto kValidTokenResponseCode = 0ul;
111 using TokenResponse = TokenEndpoint::TokenResponse;
112
113 explicit TokenProviderImpl(Settings settings,
114 std::chrono::seconds minimum_validity)
115 : minimum_validity_{minimum_validity},
116 token_(
117 TokenEndpoint(std::move(settings)).RequestAutoRefreshingToken()) {
118 }
119
121 std::string operator()() const {
122 return GetResponse().IsSuccessful()
123 ? GetResponse().GetResult().GetAccessToken()
124 : "";
125 }
126
128 ErrorResponse GetErrorResponse() const {
129 auto response = GetResponse();
130 return response.IsSuccessful() ? response.GetResult().GetErrorResponse()
131 : ErrorResponse{};
132 }
133
135 int GetHttpStatusCode() const {
136 auto response = GetResponse();
137 return response.IsSuccessful()
138 ? response.GetResult().GetHttpStatus()
139 : http::HttpStatusCode::SERVICE_UNAVAILABLE;
140 }
141
144 TokenResponse GetResponse() const {
148 std::lock_guard<std::mutex> lock(request_mutex_);
149 return token_.GetToken(minimum_validity_);
150 }
151
153 bool IsTokenResponseOK() const {
154 const auto response = GetResponse();
155 return response.IsSuccessful() &&
156 response.GetResult().GetErrorResponse().code ==
157 kValidTokenResponseCode;
158 }
159
160 private:
161 std::chrono::seconds minimum_validity_{kDefaultMinimumValidity};
162 AutoRefreshingToken token_;
163 mutable std::mutex request_mutex_;
164 };
165
166 std::shared_ptr<TokenProviderImpl> impl_;
167};
