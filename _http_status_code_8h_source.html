123 static bool IsRetryable(int http_code) {
124 if (http_code < 0)
125 return false;
126
127 switch (http_code) {
128 case HttpStatusCode::INTERNAL_SERVER_ERROR:
129 case HttpStatusCode::SERVICE_UNAVAILABLE:
130 case HttpStatusCode::TOO_MANY_REQUESTS:
131 case HttpStatusCode::BANDWIDTH_LIMIT_EXCEEDED:
132 case HttpStatusCode::REQUEST_TIMEOUT:
133 case HttpStatusCode::AUTHENTICATION_TIMEOUT:
134 case HttpStatusCode::LOGIN_TIMEOUT:
135 case HttpStatusCode::GATEWAY_TIMEOUT:
136 case HttpStatusCode::NETWORK_READ_TIMEOUT:
137 case HttpStatusCode::NETWORK_CONNECT_TIMEOUT:
138 return true;
139 default:
140 return false;
141 }
142 }

152 static olp::client::ErrorCode GetErrorCode(int http_code) {
153 // ErrorCode (negative numbers)
154 if (http_code < 0) {
155 switch (static_cast<olp::http::ErrorCode>(http_code)) {
156 case olp::http::ErrorCode::OFFLINE_ERROR:
157 case olp::http::ErrorCode::IO_ERROR:
158 return olp::client::ErrorCode::NetworkConnection;
159 case olp::http::ErrorCode::TIMEOUT_ERROR:
160 return olp::client::ErrorCode::RequestTimeout;
161 case olp::http::ErrorCode::CANCELLED_ERROR:
162 return olp::client::ErrorCode::Cancelled;
163 case olp::http::ErrorCode::AUTHORIZATION_ERROR:
164 case olp::http::ErrorCode::AUTHENTICATION_ERROR:
165 return olp::client::ErrorCode::AccessDenied;
166 case olp::http::ErrorCode::INVALID_URL_ERROR:
167 return olp::client::ErrorCode::ServiceUnavailable;
168 case olp::http::ErrorCode::NETWORK_OVERLOAD_ERROR:
169 return olp::client::ErrorCode::SlowDown;
170 case olp::http::ErrorCode::UNKNOWN_ERROR:
171 default:
172 return olp::client::ErrorCode::Unknown;
173 }
174 }
